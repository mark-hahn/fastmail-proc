# Nginx Configuration for Fastmail Host
# This has been added to /etc/nginx/conf.d/server.conf

location /fastmail {
    # Remove /fastmail prefix before proxying
    rewrite ^/fastmail(/.*)$ $1 break;
    rewrite ^/fastmail$ / break;
    
    # Proxy to the fastmail-host server
    proxy_pass http://localhost:3456;
    
    # Proxy headers
    proxy_http_version 1.1;
    proxy_set_header Upgrade $http_upgrade;
    proxy_set_header Connection 'upgrade';
    proxy_set_header Host $host;
    proxy_set_header X-Real-IP $remote_addr;
    proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
    proxy_set_header X-Forwarded-Proto $scheme;
    proxy_cache_bypass $http_upgrade;
    
    # Timeouts for long-running requests
    proxy_connect_timeout 60s;
    proxy_send_timeout 60s;
    proxy_read_timeout 60s;
}

# Configuration applied to /etc/nginx/conf.d/server.conf
# in the hahnca.com server block
# 
# To test: sudo nginx -t
# To reload: sudo systemctl reload nginx
#
# The site is accessible at: https://hahnca.com/fastmail
